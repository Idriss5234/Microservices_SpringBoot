1.CREATE DATABASE UserService;

CREATE TABLE User (
    user_id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL,
   
CREATE TABLE Preferences (
    preference_id INT AUTO_INCREMENT PRIMARY KEY,
    categories VARCHAR(256) NOT NULL,
    user_id INT,    FOREIGN KEY (user_id) REFERENCES UserService.User (user_id));

);

2.CREATE DATABASE PanierService;

CREATE TABLE Panier (
    panier_id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL, 
    item_details TEXT NOT NULL,
    disponibilité boolean,
    quantité INT NOT NULL,
    prix DECIMAL(10, 2) NOT NULL,
    a_retirer TIMESTAMP 
);

3.CREATE DATABASE CommandeService;

CREATE TABLE Commande (
    commande_id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    panier_id INT NOT NULL,
    quantité INT NOT NULL,	
    status VARCHAR(50) DEFAULT 'Pending',
    date TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

4.CREATE DATABASE PaymentService;

CREATE TABLE Paiement (
    paiement_id INT AUTO_INCREMENT PRIMARY KEY,
    commande_id INT NOT NULL, 
    user_id INT NOT NULL, 
    payment_method VARCHAR(50) NOT NULL,
    prix DECIMAL(10, 2) NOT NULL,
    payment_status VARCHAR(50) DEFAULT 'Pending',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

5.CREATE DATABASE LocationService;

CREATE TABLE Localisation_User (
    localisation_id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL, 
    latitude DECIMAL(10, 8) NOT NULL,
    longitude DECIMAL(11, 8) NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE Localisation_Partenaire (
    localisation_id INT AUTO_INCREMENT PRIMARY KEY,
    part_id INT NOT NULL, 
    latitude DECIMAL(10, 8) NOT NULL,
    longitude DECIMAL(11, 8) NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
